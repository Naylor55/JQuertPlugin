No.1预加载数据生成无限级树形菜单

		$("ele").LuTree({
			arr: arr,//节点json数据
			sign:true,//是否全部展开
			simIcon: "fa fa-folder",//叶子图标
			mouIconOpen: " fa fa-folder-open",//展开图标
			mouIconClose:"fa fa-folder",//关闭图标
			callback: function(id) {
		        //点击回调选中节点id
				console.log("你选择的id是" + id);
			}
		});
